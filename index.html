📌 Jenis Constructor di Dart

🏗️ 5 Jenis Constructor

No Jenis Constructor Sintaks
1 Default Constructor className()
2 Named Constructor className.name()
3 Parameterized Constructor className(this.param)
4 Constant Constructor const className()
5 Factory Constructor (dibahas di sesi lanjut)

🔤 Named Constructor

🚗 Class Mobil dengan Named Constructor

import 'dart:convert';

class Mobil {
  String? merk;
  int? tahun;

  // Parameterized constructor
  Mobil(this.merk, this.tahun);

  // Named constructor
  Mobil.baru() {
    merk = 'Agya';
    tahun = 2023;
  }

  // Named constructor dengan parameter
  Mobil.dariJson(Map<String, dynamic> json) {
    merk = json['merk'];
    tahun = json['tahun'];
  }

  @override
  String toString() {
    return 'Mobil{merk: $merk, tahun: $tahun}';
  }
}

void main() {
  // Constructor default
  var mobilA = Mobil('Daihatsu', 2020);
  print('mobilA: $mobilA');

  // Named constructor .baru()
  var mobilB = Mobil.baru();
  print('mobilB: $mobilB');

  // Named constructor .dariJson()
  var json = {'merk': 'Ferari', 'tahun': 2021};
  var mobilC = Mobil.dariJson(json);
  print('mobilC: $mobilC');
}

🔒 Constant Constructor

🎨 Class Warna dengan Const Constructor

class Warna {
  final int r, g, b;
  
  // Constant constructor
  const Warna(this.r, this.g, this.b);

  @override
  String toString() {
    return 'Warna(r: $r, g: $g, b: $b)';
  }
}

void main() {
  const merah = Warna(255, 0, 0);
  const merahLagi = Warna(255, 0, 0);
  const biru = Warna(0, 0, 255);
  
  print('Merah: $merah');
  print('Merah lagi: $merahLagi');
  print('Biru: $biru');
  
  // Cek apakah instance sama
  print('merah == merahLagi: ${identical(merah, merahLagi)}'); // true
}
💡 TIP: Constant constructor digunakan untuk objek immutable (tidak berubah).
Semua field harus final. Object dengan nilai yang sama akan menggunakan instance yang sama di memory.

📋 Constructor dengan Initializer List

📍 Class Titik dengan Initializer List

class Titik {
  final double x, y;

  // Initializer list untuk mengambil nilai absolut
  // sebelum constructor body dieksekusi
  Titik(double x, double y)
    : x = x.abs(),   // mengambil nilai absolut
      y = y.abs() {
    print('Titik dibuat di ($x, $y)');
  }
}

void main() {
  print('--- Titik dengan nilai positif ---');
  var titik1 = Titik(3.0, 4.0);
  
  print('\n--- Titik dengan nilai negatif ---');
  var titik2 = Titik(-5.0, 10.0);
  
  print('\n--- Titik dengan kedua nilai negatif ---');
  var titik3 = Titik(-1.5, -2.5);
  
  print('\n--- Hasil akhir ---');
  print('titik1: (${titik1.x}, ${titik1.y})');
  print('titik2: (${titik2.x}, ${titik2.y})');
  print('titik3: (${titik3.x}, ${titik3.y})');
}

🔒 ENKAPSULASI (Encapsulation)

📦 Apa itu Enkapsulasi?

Konsep: Menyembunyikan detail implementasi dari luar.

Tujuan:

  • ✅ Keamanan data
  • ✅ Kontrol akses
  • ✅ Fleksibilitas perubahan

Cara di Dart: Menggunakan underscore (_) untuk private members.

🔑 Public vs Private di Dart

🌍 Public vs 🔐 Private

class Person {
  String _name;  // Private (hanya bisa diakses dalam file yang sama)
  int age;       // Public (bisa diakses dari mana saja)
  
  Person(this._name, this.age);
  
  // Getter untuk mengakses _name
  String get name => _name;
}

void main() {
  Person person = Person("Salma", 20);
  
  print('Age (public): ${person.age}');        // ✅ Bisa (public)
  print('Name (via getter): ${person.name}');  // ✅ Bisa (via getter)
  // print(person._name);  // ❌ ERROR: private
}

📤 Getter & 📥 Setter

Getter dan Setter dengan Berbagai Cara

class Person {
  String _name;
  int age;
  
  Person(this._name, this.age);
  
  // Getter dengan keyword 'get'
  String get name => _name;
  
  // Getter dengan method biasa
  String getName() => _name;
  
  // Setter dengan keyword 'set'
  set name(String value) {
    if (value.length >= 3) {
      _name = value;
    } else {
      print('❌ Nama minimal 3 karakter');
    }
  }
  
  // Setter dengan method biasa
  void setName(String value) => _name = value;
}

void main() {
  Person person = Person("Salma", 20);
  
  print('--- Mengakses Getter ---');
  print('Via getter: ${person.name}');
  print('Via method: ${person.getName()}');
  
  print('\n--- Menggunakan Setter ---');
  person.name = 'Aqila';        // via setter
  print('Setelah setter: ${person.name}');
  
  person.setName('Nabila');     // via method
  print('Setelah method: ${person.getName()}');
  
  print('\n--- Validasi Setter ---');
  person.name = 'Ti';  // Kurang dari 3 karakter
  print('Nama tetap: ${person.name}');  // Masih "Nabila"
}

✅ Getter & Setter dengan Validasi

📦 Class Produk dengan Validasi

class Produk {
  String _nama;
  double _harga;
  
  Produk(this._nama, this._harga);
  
  // Getter
  String get nama => _nama;
  double get harga => _harga;
  
  // Setter dengan validasi
  set nama(String value) {
    if (value.length >= 3) {
      _nama = value;
    } else {
      print('❌ Gagal: Nama minimal 3 karakter.');
    }
  }
  
  set harga(double value) {
    if (value > 0) {
      _harga = value;
    } else {
      print('❌ Gagal: Harga harus > 0.');
    }
  }
}

void main() {
  var produk1 = Produk('Monitor Samsung', 3500000);

  print('--- INFORMASI PRODUK AWAL ---');
  print('Nama : ${produk1.nama}');
  print('Harga: Rp ${produk1.harga}');

  // Update dengan nilai valid
  print('\n--- UPDATE DENGAN NILAI VALID ---');
  produk1.nama = 'Keyboard Mekanik';
  produk1.harga = 1200000;
  print('Nama : ${produk1.nama}');
  print('Harga: Rp ${produk1.harga}');

  // Coba update dengan nilai tidak valid
  print('\n--- MENCOBA UPDATE DENGAN NILAI TIDAK VALID ---');
  print('Mencoba mengubah nama menjadi "TV" ( < 3 karakter)');
  produk1.nama = 'TV';
  
  print('\nMencoba mengubah harga menjadi -500000');
  produk1.harga = -500000;
  
  print('\n--- STATUS AKHIR (TIDAK BERUBAH) ---');
  print('Nama : ${produk1.nama}');   // Masih "Keyboard Mekanik"
  print('Harga: Rp ${produk1.harga}'); // Masih 1200000
}

🔐 Studi Kasus: Sistem Login Sederhana

👥 Class User dengan Method Login & Ganti Password

class User {
  String _username;
  String _password;

  User(this._username, this._password);

  String get username => _username;

  bool login(String username, String password) {
    return _username == username && _password == password;
  }

  void gantiPassword(String lama, String baru) {
    if (_password == lama) {
      if (baru.length >= 8) {
        _password = baru;
        print('✅ Password berhasil diubah.');
      } else {
        print('❌ Gagal: Password baru minimal 8 karakter.');
      }
    } else {
      print('❌ Gagal: Password lama tidak sesuai.');
    }
  }
}

void main() {
  print('=== SISTEM LOGIN SEDERHANA ===\n');
  
  // Inisialisasi user
  final user1 = User('admin_user', 'securePass123');
  print('✅ User dibuat: ${user1.username}');

  // Uji login
  print('\n--- UJI COBA LOGIN ---');
  
  // Login berhasil
  if (user1.login('admin_user', 'securePass123')) {
    print('✅ Login berhasil (username & password benar)');
  }
  
  // Login gagal (password salah)
  if (!user1.login('admin_user', 'wrongPass')) {
    print('❌ Login gagal (password salah)');
  }

  // Uji ganti password
  print('\n--- UJI COBA GANTI PASSWORD ---');
  
  // Gagal: password baru terlalu pendek
  print('Mencoba ganti password dengan "new" (terlalu pendek):');
  user1.gantiPassword('securePass123', 'new');
  
  // Berhasil: ganti password
  print('\nMencoba ganti password dengan "newSecurePass456":');
  user1.gantiPassword('securePass123', 'newSecurePass456');
  
  // Login dengan password baru
  print('\n--- LOGIN DENGAN PASSWORD BARU ---');
  if (user1.login('admin_user', 'newSecurePass456')) {
    print('✅ Login dengan password baru berhasil');
  }
}

📝 Ringkasan & Latihan

📌 Ringkasan

  1. Dart memiliki berbagai jenis constructor (default, named, constant, initializer list)
  2. Enkapsulasi melindungi data dengan private field (_field)
  3. Getter & Setter memberikan kontrol akses yang aman + validasi
  4. Constant constructor untuk objek immutable dan hemat memory
  5. Initializer list untuk memproses nilai sebelum constructor body

🏦 Latihan Mandiri

Buat kelas RekeningBank dengan:

  • Field private: _nomorRekening, _saldo, _namaPemilik
  • Getter untuk semua field
  • Method:
    • deposit(double jumlah) - menambah saldo
    • tarik(double jumlah) - mengurangi saldo
    • cekSaldo() - menampilkan saldo
  • Validasi:
    • Saldo tidak bisa negatif
    • Jumlah deposit harus > 0
    • Penarikan tidak boleh melebihi saldo

Buat program sederhana untuk simulasi transaksi.

class RekeningBank {
  String _nomorRekening;
  String _namaPemilik;
  double _saldo;
  
  RekeningBank(this._nomorRekening, this._namaPemilik, this._saldo);
  
  // Getter
  String get nomorRekening => _nomorRekening;
  String get namaPemilik => _namaPemilik;
  double get saldo => _saldo;
  
  // Method
  void deposit(double jumlah) {
    if (jumlah > 0) {
      _saldo += jumlah;
      print('✅ Deposit Rp $jumlah berhasil');
    } else {
      print('❌ Jumlah deposit harus > 0');
    }
  }
  
  void tarik(double jumlah) {
    if (jumlah > 0 && jumlah <= _saldo) {
      _saldo -= jumlah;
      print('✅ Penarikan Rp $jumlah berhasil');
    } else if (jumlah > _saldo) {
      print('❌ Saldo tidak mencukupi');
    } else {
      print('❌ Jumlah penarikan harus > 0');
    }
  }
  
  void cekSaldo() {
    print('💰 Saldo: Rp $_saldo');
  }
}
