📌 Jenis Constructor di Dart

🏗️ 5 Jenis Constructor

No Jenis Constructor Sintaks
1 Default Constructor className()
2 Named Constructor className.name()
3 Parameterized Constructor className(this.param)
4 Constant Constructor const className()
5 Factory Constructor (dibahas di sesi lanjut)

🔤 Named Constructor

🚗 Class Mobil dengan Named Constructor

import 'dart:convert';

class Mobil {
  String? merk;
  int? tahun;

  // Parameterized constructor
  Mobil(this.merk, this.tahun);

  // Named constructor .baru()
  Mobil.baru() {
    merk = 'Toyota';
    tahun = 2023;
  }

  // Named constructor .dariJson()
  Mobil.dariJson(Map<String, dynamic> json) {
    merk = json['merk'];
    tahun = json['tahun'];
  }

  @override
  String toString() {
    return 'Mobil{merk: $merk, tahun: $tahun}';
  }
}

▶️ Contoh Penggunaan

void main() {
  // Constructor default
  var mobilA = Mobil('Honda', 2020);
  print(mobilA); // Mobil{merk: Honda, tahun: 2020}

  // Named constructor .baru()
  var mobilB = Mobil.baru();
  print(mobilB); // Mobil{merk: Toyota, tahun: 2023}

  // Named constructor .dariJson()
  var json = {'merk': 'Suzuki', 'tahun': 2021};
  var mobilC = Mobil.dariJson(json);
  print(mobilC); // Mobil{merk: Suzuki, tahun: 2021}
}
import 'dart:convert';

class Mobil {
  String? merk;
  int? tahun;

  // Parameterized constructor
  Mobil(this.merk, this.tahun);

  // Named constructor
  Mobil.baru() {
    merk = 'Toyota';
    tahun = 2023;
  }

  // Named constructor dengan parameter
  Mobil.dariJson(Map<String, dynamic> json) {
    merk = json['merk'];
    tahun = json['tahun'];
  }

  @override
  String toString() {
    return 'Mobil{merk: $merk, tahun: $tahun}';
  }
}

void main() {
  print('--- Demonstrasi Class Mobil ---');

  // 1. Menggunakan constructor default
  var mobilA = Mobil('Honda', 2020);
  print('Mobil A: $mobilA');

  // 2. Menggunakan named constructor Mobil.baru()
  var mobilB = Mobil.baru();
  print('Mobil B: $mobilB');

  // 3. Menggunakan named constructor Mobil.dariJson()
  String jsonString = '{"merk": "Suzuki", "tahun": 2021}';
  Map<String, dynamic> jsonData = jsonDecode(jsonString);
  var mobilC = Mobil.dariJson(jsonData);
  print('Mobil C: $mobilC');

  // Contoh lain
  Map<String, dynamic> anotherJson = {'merk': 'Mitsubishi', 'tahun': 2022};
  var mobilD = Mobil.dariJson(anotherJson);
  print('Mobil D: $mobilD');
}

🔒 Constant Constructor

🎨 Class Warna dengan Const Constructor

class Warna {
  final int r, g, b;
  
  // Constant constructor
  const Warna(this.r, this.g, this.b);

  @override
  String toString() {
    return 'Warna(r: $r, g: $g, b: $b)';
  }
}
💡 TIP: Constant constructor digunakan untuk objek immutable (tidak berubah).
Semua field harus final.

▶️ Contoh & Keuntungan

void main() {
  const merah = Warna(255, 0, 0);
  const merahLagi = Warna(255, 0, 0);
  
  print(merah);      // Warna(r: 255, g: 0, b: 0)
  print(merahLagi);  // Warna(r: 255, g: 0, b: 0)
  
  // Menghemat memory: merah dan merahLagi
  // menggunakan INSTANCE YANG SAMA!
}

📋 Constructor dengan Initializer List

📍 Class Titik dengan Initializer List

class Titik {
  final double x, y;

  // Initializer list untuk validasi/modifikasi
  // sebelum constructor body dieksekusi
  Titik(double x, double y)
    : x = x.abs(),   // mengambil nilai absolut
      y = y.abs() {
    print('Titik dibuat di ($x, $y)');
  }
}

▶️ Contoh Penggunaan

void main() {
  // Nilai positif
  var titik1 = Titik(3.0, 4.0);
  // Output: Titik dibuat di (3.0, 4.0)
  
  // Nilai negatif akan di-absolutkan
  var titik2 = Titik(-5.0, 10.0);
  // Output: Titik dibuat di (5.0, 10.0)
  
  var titik3 = Titik(-1.5, -2.5);
  // Output: Titik dibuat di (1.5, 2.5)
}

🔒 ENKAPSULASI (Encapsulation)

📦 Apa itu Enkapsulasi?

Konsep: Menyembunyikan detail implementasi dari luar.

Tujuan:

  • ✅ Keamanan data
  • ✅ Kontrol akses
  • ✅ Fleksibilitas perubahan

Cara di Dart: Menggunakan underscore (_) untuk private members.

🔑 Public vs Private di Dart

🌍 Public (Default)

class Person {
  String name;  // Public
  int age;      // Public
  
  Person(this.name, this.age);
}

Dapat diakses dari mana saja.

🔐 Private (dengan _ )

class Person {
  String _name;  // Private
  int age;       // Public
  
  Person(this._name, this.age);
}

Hanya dapat diakses dalam file yang sama.

void main() {
  Person person = Person("Salma", 20);
  print(person.age);       // ✅ Bisa (public)
  // print(person._name);  // ❌ ERROR karena private
}

📤 Getter – Mengakses Data Private

Getter: 2 Cara Penulisan

class Person {
  String _name;
  int age;
  
  Person(this._name, this.age);
  
  // Cara 1: Getter dengan keyword 'get'
  String get getName1 {
    return _name;
  }
  
  // Cara 2: Method biasa
  String getName() {
    return _name;
  }
}

▶️ Penggunaan

void main() {
  Person person = Person("Salma", 20);
  
  print(person.getName());   // Memanggil method
  print(person.getName1);    // Memanggil getter (tanpa kurung)
}

📥 Setter – Memodifikasi Data Private

Setter dengan Validasi

class Person {
  String _name;
  int age;
  
  Person(this._name, this.age);
  
  // Setter dengan keyword 'set'
  set setName1(String name) {
    _name = name;
  }

  // Setter dengan method biasa
  void setName(String name) {
    _name = name;
  }

  String getName() {
    return _name;
  }
}

void main() {
  Person person = Person("Salma", 20);
  
  person.setName('Aqila');      // via method
  print(person.getName());      // Aqila
  
  person.setName1 = 'Nabila';   // via setter (tanpa kurung, pakai =)
  print(person.getName());      // Nabila
}

✅ Getter & Setter Lengkap dengan Validasi

📦 Class Produk dengan Validasi

class Produk {
  String _nama;
  double _harga;
  
  Produk(this._nama, this._harga);
  
  // Getter
  String get nama => _nama;
  double get harga => _harga;
  
  // Setter dengan validasi
  set nama(String value) {
    if (value.length >= 3) {
      _nama = value;
    } else {
      print('❌ Gagal: Nama minimal 3 karakter.');
    }
  }
  
  set harga(double value) {
    if (value > 0) {
      _harga = value;
    } else {
      print('❌ Gagal: Harga harus > 0.');
    }
  }
}
class Produk {
  String _nama;
  double _harga;
  
  Produk(this._nama, this._harga);
  
  // Getter
  String get nama => _nama;
  double get harga => _harga;
  
  // Setter dengan validasi
  set nama(String value) {
    if (value.length >= 3) {
      _nama = value;
    } else {
      print('❌ Gagal memperbarui nama: Nama harus memiliki minimal 3 karakter.');
    }
  }
  
  set harga(double value) {
    if (value > 0) {
      _harga = value;
    } else {
      print('❌ Gagal memperbarui harga: Harga harus lebih besar dari 0.');
    }
  }
}

void main() {
  var produk1 = Produk('Monitor Samsung', 3500.0);

  print('--- Informasi Produk Awal ---');
  print('Nama: ${produk1.nama}');
  print('Harga: Rp ${produk1.harga}');

  // Update dengan nilai valid
  print('\n--- Memperbarui Produk ---');
  produk1.nama = 'Keyboard Mekanik';
  produk1.harga = 1200.0;
  print('Nama: ${produk1.nama}');
  print('Harga: Rp ${produk1.harga}');

  // Coba update dengan nilai tidak valid
  print('\n--- Menguji Validasi ---');
  produk1.nama = 'TV';        // ❌ Gagal ( < 3 karakter)
  produk1.harga = -100.0;     // ❌ Gagal ( < 0)
  
  print('\n--- Status Akhir ---');
  print('Nama: ${produk1.nama}');   // Masih "Keyboard Mekanik"
  print('Harga: Rp ${produk1.harga}'); // Masih 1200.0
}

🧪 Praktik 2 – Enkapsulasi Lengkap

👤 Buat kelas User dengan:

  • Field private: _username, _password, _email
  • Getter untuk semua field
  • Setter dengan validasi:
    • Username minimal 5 karakter
    • Password minimal 8 karakter
    • Email harus mengandung '@'

🔐 Studi Kasus: Sistem Login Sederhana

👥 Class User dengan Method Login & Ganti Password

class User {
  String _username;
  String _password;

  User(this._username, this._password);

  String get username => _username;

  bool login(String username, String password) {
    return _username == username && _password == password;
  }

  void gantiPassword(String lama, String baru) {
    if (_password == lama) {
      if (baru.length >= 8) {
        _password = baru;
        print('✅ Password berhasil diubah.');
      } else {
        print('❌ Gagal: Password baru minimal 8 karakter.');
      }
    } else {
      print('❌ Gagal: Password lama tidak sesuai.');
    }
  }
}
class User {
  String _username;
  String _password;

  User(this._username, this._password);

  String get username => _username;

  bool login(String username, String password) {
    return _username == username && _password == password;
  }

  void gantiPassword(String lama, String baru) {
    if (_password == lama) {
      if (baru.length >= 8) {
        _password = baru;
        print('✅ Password berhasil diubah.');
      } else {
        print('❌ Gagal: Password baru minimal 8 karakter.');
      }
    } else {
      print('❌ Gagal: Password lama tidak sesuai.');
    }
  }
}

void main() {
  print('--- Inisialisasi User ---');
  final user1 = User('admin_user', 'securePass123');
  print('User: ${user1.username}');

  // Uji login
  print('\n--- Uji Login ---');
  if (user1.login('admin_user', 'securePass123')) {
    print('✅ Login berhasil');
  } else {
    print('❌ Login gagal');
  }

  // Uji ganti password
  print('\n--- Ganti Password ---');
  user1.gantiPassword('securePass123', 'newPass456'); // ❌ (hanya 8 karakter? hitung sendiri)
  user1.gantiPassword('securePass123', 'newSecurePass456'); // ✅
  
  // Login dengan password baru
  if (user1.login('admin_user', 'newSecurePass456')) {
    print('✅ Login dengan password baru berhasil');
  }
}
